\section*{Problem 1}

\textit{If $f(x)=x+\sqrt{2-x}$ and $g(x)=u+\sqrt{2-u}$, is it true that $f=g$?}

\subsection*{Solution}

True


\section*{Problem 2}

\textit{If $$f(x)=\frac{x^2-x}{x-1} \quad\mathrm{and}\quad g(x)=x$$ is it true that $f=g$?}

\subsection*{Solution}

False
